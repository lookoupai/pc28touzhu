## 4.6 复现脚本

低命阈值回测核心逻辑（来源仓）：

```python
import sqlite3, statistics
conn = sqlite3.connect('/www/wwwroot/AITradingSimulator/pc28_predictor.db')
cur = conn.cursor()

def hits(pid, metric):
    cur.execute(f'SELECT hit_{metric} FROM predictions WHERE predictor_id=? AND status="settled" AND hit_{metric} IS NOT NULL ORDER BY created_at', (pid,))
    return [r[0] for r in cur.fetchall()]

def eval_thr(seq, trigger, horizon=40, win=100, cooldown=20, direction='low'):
    res, last = [], -999
    for i in range(win, len(seq)-horizon):
        rate = sum(seq[i-win:i])/win*100
        if i-last < cooldown: continue
        cond = rate <= trigger if direction=='low' else rate >= trigger
        if cond:
            fut = sum(seq[i:i+horizon])/horizon*100
            res.append((rate, fut)); last = i
    return res

# 例：组合马尔可夫V1 (pred#8) combo 旧阈值<20
seq = hits(8, 'combo')
arr = eval_thr(seq, 20, direction='low')
print('triggers=%d, fut40=%.1f%%' % (len(arr), statistics.mean(r[1] for r in arr)))
```
